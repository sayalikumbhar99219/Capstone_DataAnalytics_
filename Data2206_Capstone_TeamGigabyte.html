#Tukey Method

# Import required libraries
from collections import Counter

# Outlier detection 
def detect_outliers(df,n,features):
    
    outlier_indices = []
    
    # iterate over features(columns)
    for col in features:
        # 1st quartile (25%)
        Q1 = np.percentile(df[col], 25)
        # 3rd quartile (75%)
        Q3 = np.percentile(df[col],75)
        # Interquartile range (IQR)
        IQR = Q3 - Q1
        
        # outlier step
        outlier_step = 1.5 * IQR
        
        # Determine a list of indices of outliers for feature col
        outlier_list_col = df[(df[col] < Q1 - outlier_step) | (df[col] > Q3 + outlier_step )].index
        
        # append the found outlier indices for col to the list of outlier indices 
        outlier_indices.extend(outlier_list_col)
        
    # select observations containing more than 2 outliers
    outlier_indices = Counter(outlier_indices)        
    multiple_outliers = list( k for k, v in outlier_indices.items() if v > n )
    
    return multiple_outliers

# Function to plot learning curves
def plot_learning_curves(model, model_name, X_train, y_train):
    # Create a pipeline with standard scaling and model
    pipe = Pipeline([
        ('scaler', StandardScaler()),
        ('model', model)
    ])

    # Compute learning curves
    train_sizes, train_scores, test_scores = learning_curve(
        estimator=pipe,
        X=X_train,
        y=y_train,
        train_sizes=np.linspace(0.1, 1.0, 10),
        cv=RepeatedKFold(n_splits=10, n_repeats=5, random_state=42),
        scoring='neg_root_mean_squared_error'
    )
    
    # Calculate mean and std of train and test scores
    train_mean = np.mean(-train_scores, axis=1)
    test_mean = np.mean(-test_scores, axis=1)
    train_std = np.std(-train_scores, axis=1)
    test_std = np.std(-test_scores, axis=1)

    # Plot Learning Curves
    plt.figure(figsize=(10, 6))
    plt.plot(train_sizes, train_mean, label=f'{model_name} - Train', color='blue')
    plt.plot(train_sizes, test_mean, label=f'{model_name} - Test', color='green')
    plt.fill_between(train_sizes, train_mean - train_std, train_mean + train_std, color='blue', alpha=0.2)
    plt.fill_between(train_sizes, test_mean - test_std, test_mean + test_std, color='green', alpha=0.2)
    
    plt.title(f'Learning Curve: {model_name}')
    plt.xlabel('Training Size')
    plt.ylabel('Negative RMSE')
    plt.legend()
    plt.grid(True)
    plt.show()

from sklearn.model_selection import GridSearchCV, RepeatedKFold
from sklearn.metrics import classification_report, confusion_matrix  

def Gridsearch_cv(model, params, X_train, y_train, X_test, y_test, data, pipeline):
    # Cross-validation setup
    cv2 = RepeatedKFold(n_splits=10, n_repeats=5, random_state=100)

    # GridSearchCV
    gs_clf = GridSearchCV(model, params, cv=cv2, scoring='recall_weighted')
    gs_clf.fit(X_train, y_train)
    model = gs_clf.best_estimator_

    # Use the best model to predict on test data
    y_pred = model.predict(X_test)

    # Identify the best parameters
    bestpara = str(gs_clf.best_params_)

    # Output model information
    print('\nOptimized Model')
    print('\nModel Name:', str(model.named_steps['clf']))
    print('\n')

    # Feature Importance (for tree-based models)
    try:
        print('Feature Importances')
        for name, score in zip(list(data), gs_clf.best_estimator_.named_steps['clf'].feature_importances_):
            print(name, round(score, 2))
    except AttributeError:
        print("Feature importance is not available for this model.")

    # Dynamically determine target names from y_train or y_test
    unique_classes = sorted(set(y_train))  # Ensure correct order
    print('\nBest Parameters:', bestpara)
    print('\n', confusion_matrix(y_test, y_pred))  
    print('\n', classification_report(y_test, y_pred, target_names=[str(cls) for cls in unique_classes]))
